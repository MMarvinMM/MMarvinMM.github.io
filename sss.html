function exportPDF() {
    map2Img(mapView);
    const scale = window.devicePixelRatio;
    var threeline = document.getElementById("three-line");
    threeline.style.display = "none";
    var container = document.getElementById("container");
    html2canvas(container, {
        scale: scale,
        logging: false,
        background: '#FFF',
        x: 0,
        // y: window.pageXOffset,
        y: document.getElementById("mapview").offsetTop,
        useCORS: true,
        foreignObjectRendering: true,
        allowTaint: true,//允许跨域（图片跨域相关）
        taintTest: false,
        dpi: 300,
        onrendered: function (canvas) {
            //返回图片URL，参数：图片格式和清晰度(0-1)
            var pageData = canvas.toDataURL('image/jpeg', 1.0);
            //方向默认竖直，尺寸ponits，格式a4【595.28,841.89]
            var pdf = new jsPDF('l', 'pt', 'a4');
            //需要dataUrl格式
            pdf.addImage(pageData, 'JPEG', 40, 60, 761.89, 761.89 / 761.89 * 475.28);
            pdf.save('content.pdf');
        }
    })
    threeline.style.display = "block";
    // $mapview.removeChild(map2Img)
}
